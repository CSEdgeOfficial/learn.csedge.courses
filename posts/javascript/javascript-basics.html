        //! primitive data types(string, number, boolean)
        let FirstName = "sonu";
        let age = 21;
        let isMarried = false;

        for (let i = 0; i <= 100; i++) {
            res += i;
        }
        console.log(res)

            return 'Hello, ' + name + '!';
        }

        console.log(greet('World'));
